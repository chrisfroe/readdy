PROJECT(runUnitTests_singlecpu)

FIND_PACKAGE(Threads REQUIRED)

LIST(APPEND READDY_SINGLECPU_TEST_SOURCES SingleCPUTestMain.cpp)
LIST(APPEND READDY_SINGLECPU_TEST_SOURCES SingleCPUTestParticleData.cpp)
LIST(APPEND READDY_SINGLECPU_TEST_SOURCES SingleCPUTestReactions.cpp)
LIST(APPEND READDY_SINGLECPU_TEST_SOURCES SingleCPUTestNeighborList.cpp)

ADD_EXECUTABLE(${PROJECT_NAME} ${READDY_SINGLECPU_TEST_SOURCES} ${GOOGLETEST_INCLUDE} ${ReaDDy_INCLUDE_DIRS} ${SINGLECPU_INCLUDE_DIR} ${TESTING_INCLUDE_DIR})
TARGET_INCLUDE_DIRECTORIES(${PROJECT_NAME} PUBLIC ${ReaDDy_INCLUDE_DIRS} ${SINGLECPU_INCLUDE_DIR} ${GOOGLETEST_INCLUDE} ${GOOGLEMOCK_INCLUDE} ${gtest_SOURCE_DIR} ${TESTING_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${PROJECT_NAME} ReaDDy ReaDDy_kernel_singlecpu gtest gtest_main gmock gmock_main ${CMAKE_THREAD_LIBS_INIT})

ADD_TEST(NAME readdy-tests-singlecpu COMMAND ${PROJECT_NAME})

IF (READDY_INSTALL_UNIT_TEST_EXECUTABLE)
    INSTALL(TARGETS ${PROJECT_NAME}
            RUNTIME DESTINATION bin
            LIBRARY DESTINATION lib
            ARCHIVE DESTINATION lib)
ENDIF (READDY_INSTALL_UNIT_TEST_EXECUTABLE)
