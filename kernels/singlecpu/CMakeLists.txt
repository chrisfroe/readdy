PROJECT(ReaDDy2_kernel_singlecpu CXX)

# sources
LIST(APPEND SINGLECPU_SOURCES SingleCPUKernel.cpp)

# (private) headers
LIST(APPEND SINGLECPU_HEADERS SingleCPUKernel.h)

# create library
ADD_LIBRARY(${PROJECT_NAME} SHARED ${SINGLECPU_SOURCES} ${SINGLECPU_HEADERS} ${ReaDDy2_INCLUDE_DIRS})

# includes
TARGET_INCLUDE_DIRECTORIES(${PROJECT_NAME} PUBLIC ${ReaDDy2_INCLUDE_DIRS} ${Boost.DLL_INCLUDE_DIRS})

# link
TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${CMAKE_DL_LIBS} ReaDDy2)

SET_TARGET_PROPERTIES(${PROJECT_NAME} PROPERTIES
    POSITION_INDEPENDENT_CODE ON
    LINK_FLAGS "${EXTRA_LINK_FLAGS}"
    COMPILE_FLAGS "${EXTRA_COMPILE_FLAGS}"
)

INSTALL(TARGETS ${PROJECT_NAME} LIBRARY DESTINATION lib/readdy_plugins)
