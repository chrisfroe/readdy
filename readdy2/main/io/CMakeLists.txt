PROJECT(ReaDDy2_io CXX)

LIST(APPEND IO_INCLUDE_DIRS "${READDY_GLOBAL_INCLUDE_DIR}/io" ${Boost_INCLUDE_DIRS})
IF(READDY_BUILD_PYTHON_WRAPPER)
    LIST(APPEND IO_INCLUDE_DIRS ${PYTHON_INCLUDE_DIRS})
ENDIF()
INCLUDE_DIRECTORIES(${IO_INCLUDE_DIRS})

# sources
LIST(APPEND READDY_IO_SOURCES ReaDDy_IO.cpp)

# library
ADD_LIBRARY(${PROJECT_NAME} STATIC ${READDY_IO_SOURCES} ${READDY_IO_HEADERS} ${IO_INCLUDE_DIRS})

# Required on Unix OS family to be able to be linked into shared libraries.
SET_TARGET_PROPERTIES(${PROJECT_NAME} PROPERTIES POSITION_INDEPENDENT_CODE ON)

# link
TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${READDY_LIBRARIES})

# expose includes
SET(${PROJECT_NAME}_INCLUDE_DIRS ${IO_INCLUDE_DIRS} CACHE INTERNAL "${PROJECT_NAME}: Include directories" FORCE)
