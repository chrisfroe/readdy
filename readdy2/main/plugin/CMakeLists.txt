PROJECT(ReaDDy2_plugin)

# TODO

# sources
LIST(APPEND READDY_PLUGIN_SOURCES
        KernelProvider.cpp)

# library
ADD_LIBRARY(${PROJECT_NAME} STATIC ${READDY_PLUGIN_SOURCES} "${READDY_GLOBAL_INCLUDE_DIR}/plugin")

# expose includes
SET(${PROJECT_NAME}_INCLUDE_DIRS "${READDY_GLOBAL_INCLUDE_DIR}/plugin;${ReaDDy2_common_INCLUDE_DIRS}" CACHE INTERNAL "${PROJECT_NAME}: Include directories" FORCE)

# includes
TARGET_INCLUDE_DIRECTORIES(${PROJECT_NAME} PUBLIC ${ReaDDy2_plugin_INCLUDE_DIRS})

# Required on Unix OS family to be able to be linked into shared libraries.
SET_TARGET_PROPERTIES(${PROJECT_NAME} PROPERTIES POSITION_INDEPENDENT_CODE ON)

# link
TARGET_LINK_LIBRARIES(${PROJECT_NAME} ReaDDy2_common)

# use c++11
set_property (TARGET ${PROJECT_NAME} PROPERTY COMPILE_FLAGS -std=c++11)
