# IO -> READDY_IO
ADD_SUBDIRECTORY(io)

# sources and headers
LIST(APPEND READDY_SOURCES Simulation.cpp)

# create library
ADD_LIBRARY(${READDY_LIBRARY_NAME} STATIC ${READDY_SOURCES} ${READDY_HEADERS} ${READDY_IO_HEADERS})
# link
TARGET_LINK_LIBRARIES(${READDY_LIBRARY_NAME} ${READDY_LIBRARIES} READDY_IO)
# use c++11
set_property (TARGET ${READDY_LIBRARY_NAME} PROPERTY COMPILE_FLAGS -std=c++11)
